% Formato de capítulos
% ---------------------------------
\titleformat{\chapter}[display]
  {\filcenter\chapterfont\bfseries}
  {\fontsize{26}{30}\selectfont\MakeUppercase{Capítulo \Roman{chapter}}}
  {0.9ex}
  {\fontsize{22}{26}\selectfont\MakeUppercase}

  {\filcenter\chapterfont\bfseries}
  {}
  {0pt}
  {\fontsize{22}{26}\selectfont\MakeUppercase}

  {\sectionfont\bfseries\Large}
  {\thesection}{0.75em}{}

\titleformat{\subsection}
  {\sectionfont\bfseries\large}
  {\thesubsection}{0.75em}{}

\titleformat{\subsubsection}
  {\sectionfont\bfseries\normalsize}
  {\thesubsubsection}{0.75em}{}

\titlespacing*{\section}{0pt}{2.5ex plus 0.5ex minus 0.2ex}{1.2ex}
\titlespacing*{\subsection}{0pt}{2ex plus 0.4ex minus 0.2ex}{1ex}
